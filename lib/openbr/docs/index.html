REST API

  • path: '/openbr/faces'
    method: 'GET'
    response: [{name:'thename', fileName: 'filename'}, {name:'thename', fileName: 'filename'}, ...]
  • path: '/openbr/faces/{filename}'
    method: 'GET'
    response: filename (example: '/openbr/faces/bruce_willis.jpg')
  • path: '/openbr/faces',
    method: 'POST'
    payload: {name:'thename', image:'base64imagestring...'}
    response: {name:'thename'})
  • path: '/openbr/faces/{name}',
    method: 'DELETE'
    response: {deleted:'name', left:[{name:'thename', fileName: 'filename'}, {name:'thename', fileName: 'filename'}...]}
  • path: '/openbr/identify',
    method: 'POST'
    payload: {image:'base64imagestring...'}
    response: [{name: 'thename', score: score}, ...]
  • path: '/openbr/compare',
    method: 'POST'
    payload: {image1:'base64imagestring...', image2:'base64imagestring...'}
    response: score
